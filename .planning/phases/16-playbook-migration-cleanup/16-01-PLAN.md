  <action>
Update the getPlaybook() function (lines 191-206) to explicitly handle both 'data_loss' and 'datenverlust' strings in separate switch cases, both returning DATA_LOSS_PLAYBOOK. The default case should NOT silently return RANSOMWARE_PLAYBOOK — instead, log a warning and explicitly state the fallback.

Current code (lines 191-206):
```typescript
export const getPlaybook = (type: string): Playbook => {
  switch (type) {
    case 'ransomware':
      return RANSOMWARE_PLAYBOOK
    case 'phishing':
      return IMPORTED_PHISHING_PLAYBOOK
    case 'ddos':
      return DDOS_PLAYBOOK
    case 'data_loss':
      return DATA_LOSS_PLAYBOOK
    case 'datenverlust':
      return DATA_LOSS_PLAYBOOK
    default:
      return RANSOMWARE_PLAYBOOK  // <-- SILENT FALLBACK
  }
}
```

Fix: Keep the structure above but add a console.warn() in the default case to log when an unknown type is provided, making the fallback explicit and visible in logs:

```typescript
export const getPlaybook = (type: string): Playbook => {
  switch (type) {
    case 'ransomware':
      return RANSOMWARE_PLAYBOOK
    case 'phishing':
      return IMPORTED_PHISHING_PLAYBOOK
    case 'ddos':
      return DDOS_PLAYBOOK
    case 'data_loss':
      return DATA_LOSS_PLAYBOOK
    case 'datenverlust':
      return DATA_LOSS_PLAYBOOK
    default:
      console.warn(`[getPlaybook] Unknown incident type "${type}", falling back to ransomware playbook`)
      return RANSOMWARE_PLAYBOOK
  }
}
```

This ensures:
- Both 'data_loss' (API) and 'datenverlust' (wizard internal) map to same playbook
- Default fallback is visible in logs (not silent)
- Data Loss incidents never get wrong playbook due to string mismatch
