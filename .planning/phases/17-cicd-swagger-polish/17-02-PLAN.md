Create `src/app/api/swagger/route.ts` that serves Swagger UI as HTML.

This route:
1. Triggers on GET /api/swagger (App Router convention)
2. Generates Swagger UI HTML with swagger-ui-dist CDN
3. Passes swaggerSpec from src/lib/swagger.ts
4. Customizes CSS with SIAG branding (#CC0033 red)
5. Returns HTML with Content-Type: text/html
6. Handles /api/swagger/openapi.json sub-route for raw spec

File location: `src/app/api/swagger/route.ts`

```typescript
/**
 * /api/swagger — OpenAPI Documentation UI
 *
 * GET  /api/swagger  — Interactive Swagger UI (HTML page)
 * GET  /api/swagger/openapi.json  — Raw OpenAPI spec (JSON)
 *
 * Accessible in production at https://siag-incident-assistant.vercel.app/api/swagger
 */

import { NextResponse, NextRequest } from 'next/server';
import { swaggerSpec } from '@/lib/swagger';

/**
 * Render Swagger UI as HTML or return raw OpenAPI spec
 */
export async function GET(request: NextRequest): Promise<Response> {
  const pathname = request.nextUrl.pathname;

  // Check if requesting the raw OpenAPI spec
  if (pathname.endsWith('/openapi.json')) {
    return NextResponse.json(swaggerSpec, {
      headers: {
        'Content-Type': 'application/json',
        'Cache-Control': 'public, max-age=3600', // Cache spec for 1 hour
      },
    });
  }

  // Generate Swagger UI HTML
  try {
    const swaggerUiHtml = generateSwaggerHTML();

    return new NextResponse(swaggerUiHtml, {
      status: 200,
      headers: {
        'Content-Type': 'text/html; charset=utf-8',
        'Cache-Control': 'public, max-age=3600', // Cache HTML for 1 hour
      },
    });
  } catch (error) {
    console.error('[GET /api/swagger] Error generating Swagger UI:', error);
    return NextResponse.json(
      { error: 'Failed to generate Swagger UI' },
      { status: 500 }
    );
  }
}

/**
 * Generate Swagger UI HTML with CDN assets
 * 
 * Uses Swagger UI assets from CDN (swagger-ui-dist@4) to avoid bundling
 * Customizes styling with SIAG branding (#CC0033)
 */
function generateSwaggerHTML(): string {
  const specUrl = '/api/swagger/openapi.json';

  return `
    <!DOCTYPE html>
    <html lang="en">
    <head>
      <meta charset="utf-8" />
      <meta name="viewport" content="width=device-width, initial-scale=1" />
      <meta
        name="description"
        content="SIAG Incident Management API Documentation"
      />
      <title>SIAG Incident API Docs</title>
      
      <!-- Swagger UI CSS from CDN -->
      <link rel="stylesheet" href="https://unpkg.com/swagger-ui-dist@4/swagger-ui.css" />
      <style>
        html {
          box-sizing: border-box;
          overflow: -moz-scrollbars-vertical;
          overflow-y: scroll;
        }
        *,
        *:before,
        *:after {
          box-sizing: inherit;
        }
        body {
          margin: 0;
          padding: 0;
        }

        /* SIAG branding */
        .swagger-ui .topbar {
          background-color: #CC0033;
        }
        .swagger-ui .information .title {
          color: #CC0033;
          font-weight: 700;
        }
        .swagger-ui .btn-authorize {
          background-color: #CC0033;
          border-color: #CC0033;
        }
        .swagger-ui .btn-authorize:hover {
          background-color: #990026;
          border-color: #990026;
        }
        .swagger-ui .model-box {
          background: rgba(204, 0, 51, 0.05);
          border: 1px solid #CC0033;
        }
        .swagger-ui .scheme-container {
          background: #fafafa;
          margin: 0 0 20px;
          padding: 30px;
          border-radius: 4px;
          border: 2px solid #CC0033;
        }
      </style>
    </head>
    <body>
      <div id="swagger-ui"></div>

      <!-- Swagger UI bundle from CDN -->
      <script src="https://unpkg.com/swagger-ui-dist@4/swagger-ui-bundle.js" crossorigin></script>
      <script src="https://unpkg.com/swagger-ui-dist@4/swagger-ui-standalone-preset.js" crossorigin></script>

      <script>
        window.onload = () => {
          const ui = SwaggerUIBundle({
            url: '${specUrl}',
            dom_id: '#swagger-ui',
            presets: [
              SwaggerUIBundle.presets.apis,
              SwaggerUIStandalonePreset
            ],
            plugins: [
              SwaggerUIBundle.plugins.DownloadUrl
            ],
            layout: "StandaloneLayout",
            defaultModelsExpandDepth: 1,
            docExpansion: "list",
            displayOperationId: true,
            persistAuthorization: true,
            filter: true,
            showExtensions: false,
            supportedSubmitMethods: [
              'get', 'post', 'put', 'delete', 'patch', 'options', 'trace'
            ]
          });
          window.ui = ui;
        };
      </script>
    </body>
    </html>
  `;
}
```
