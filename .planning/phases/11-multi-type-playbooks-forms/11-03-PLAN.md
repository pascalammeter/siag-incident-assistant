# Plan 11-03 — Form Validation: onBlur, Error Display, Helper Text

## Goal

Enhance form validation across all wizard steps with real-time onBlur validation, inline error messages, helper text for complex inputs, and visual error feedback (red borders, asterisks on required fields).

## Success Criteria

1. ✅ Form validation triggered on field blur (not just submit)
2. ✅ Error messages display immediately below field when validation fails
3. ✅ Error text in red (#CC0033), field border changes to red on error
4. ✅ Required fields marked with red asterisk (*) in label
5. ✅ Helper text appears below complex inputs (multi-select, date) explaining constraints
6. ✅ Validation cleared when field corrected and re-blurred
7. ✅ Save button disabled during API request (handled in 11-04 but tested here)
8. ✅ All form components use consistent error display pattern
9. ✅ 80%+ test coverage on validation logic

### 2. Create onBlur Validation Hook (src/hooks/useFormValidation.ts)

```typescript
export const useFormValidation = (schema: z.ZodSchema) => {
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [touched, setTouched] = useState<Record<string, boolean>>({});

  const validateField = (fieldName: string, value: any) => {
    try {
      schema.pick({ [fieldName]: true }).parse({ [fieldName]: value });
      setErrors(prev => ({ ...prev, [fieldName]: '' }));
    } catch (error) {
      if (error instanceof z.ZodError) {
        setErrors(prev => ({
          ...prev,
          [fieldName]: error.errors[0]?.message || 'Invalid input'
        }));
      }
    }
  };

  const handleBlur = (fieldName: string) => (event: FocusEvent) => {
    setTouched(prev => ({ ...prev, [fieldName]: true }));
    const target = event.target as HTMLInputElement;
    validateField(fieldName, target.value);
  };

  return { errors, touched, handleBlur, validateField };
};
```

### 3. Create FormField Component (src/components/FormField.tsx)

Wrapper component for all form inputs:

```typescript
interface FormFieldProps {
  label: string;
  name: string;
  type?: string;
  required?: boolean;
  helperText?: string;
  error?: string;
  onBlur?: (e: FocusEvent) => void;
  children?: React.ReactNode; // for custom input types
}

export const FormField: React.FC<FormFieldProps> = ({
  label, name, type = 'text', required, helperText, error, onBlur, children
}) => (
  <div className="form-group">
    <label htmlFor={name}>
      {label}
      {required && <span className="text-red-600 ml-1">*</span>}
    </label>
    {children || (
      <input
        id={name}
        name={name}
        type={type}
        onBlur={onBlur}
        className={`input ${error ? 'border-red-600' : ''}`}
      />
    )}
    {error && touched[name] && (
      <div className="text-red-600 text-sm mt-1">{error}</div>
    )}
    {helperText && !error && (
      <div className="text-gray-500 text-sm mt-1">{helperText}</div>
    )}
  </div>
);
```

### 4. Update StepForm Component (src/components/StepForm.tsx)

```typescript
export const StepForm: React.FC<StepFormProps> = ({ step, onNext }) => {
  const { errors, touched, handleBlur, validateForm } = useFormValidation(stepSchemas[step.number]);

  const handleSubmit = async (e: FormEvent) => {
    e.preventDefault();
    if (!validateForm()) return; // validation failed
    onNext();
  };

  return (
    <form onSubmit={handleSubmit}>
      {/* Form fields using FormField component */}
      <FormField
        label="Description"
        name="description"
        required
        helperText="Provide a detailed description of the incident"
        error={errors.description}
        onBlur={handleBlur('description')}
      />
      <button type="submit">Next</button>
    </form>
  );
};
```
