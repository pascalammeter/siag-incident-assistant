**Test Case 4: Soft-delete sets deletedAt timestamp**
```typescript
it('should soft-delete incident with deletedAt timestamp', async () => {
  const createResponse = await request(app)
    .post('/api/incidents')
    .send({
      incident_type: 'phishing',
      severity: 'medium',
    })
    .expect(201);

  const incidentId = createResponse.body.id;

  const deleteResponse = await request(app)
    .delete(`/api/incidents/${incidentId}`)
    .expect(204);

  // Verify deletedAt is set in database
  const deletedIncident = await prisma.incident.findUnique({
    where: { id: incidentId },
  });

  expect(deletedIncident.deletedAt).not.toBeNull();
  expect(deletedIncident.deletedAt).toBeInstanceOf(Date);
  expect(Math.abs(Date.now() - deletedIncident.deletedAt.getTime())).toBeLessThan(1000);
});
```

```typescript
it('should exclude soft-deleted incidents from list query', async () => {
  // Create 2 active, 1 soft-deleted
  const incident1 = await request(app)
    .post('/api/incidents')
    .send({ incident_type: 'ransomware', severity: 'critical' })
    .expect(201);

  const incident2 = await request(app)
    .post('/api/incidents')
    .send({ incident_type: 'phishing', severity: 'high' })
    .expect(201);

  const incident3 = await request(app)
    .post('/api/incidents')
    .send({ incident_type: 'ddos', severity: 'medium' })
    .expect(201);

  // Delete one
  await request(app)
    .delete(`/api/incidents/${incident3.body.id}`)
    .expect(204);

  // List should have 2, not 3
  const listResponse = await request(app)
    .get('/api/incidents')
    .expect(200);

  expect(listResponse.body.data.length).toBe(2);
  expect(listResponse.body.total).toBe(2);
  expect(listResponse.body.data.map(i => i.id)).toContain(incident1.body.id);
  expect(listResponse.body.data.map(i => i.id)).toContain(incident2.body.id);
  expect(listResponse.body.data.map(i => i.id)).not.toContain(incident3.body.id);
});
```
