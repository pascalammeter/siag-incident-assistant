    Build deadlines array conditionally:
    ```typescript
    const deadlines: Array<{ law: string; deadline: Date | null; badge?: string; text?: string; note?: string }> = []

    if (form.watch('kritischeInfrastruktur') === 'ja') {
      if (erkennungszeitpunkt) {
        deadlines.push({ law: 'ISG — NCSC-Meldung', deadline: computeDeadline(erkennungszeitpunkt, 24), badge: '24h' })
      }
    }
    if (form.watch('personendatenBetroffen') === 'ja') {
      deadlines.push({ law: 'DSG/DSGVO — FDPIC-Meldung', deadline: null, text: 'So schnell wie moeglich', note: 'Keine feste Frist — unverzueglich nach Kenntnis.' })
    }
    if (form.watch('reguliertesUnternehmen') === 'ja') {
      if (erkennungszeitpunkt) {
        deadlines.push(
          { law: 'FINMA — Informelle Meldung', deadline: computeDeadline(erkennungszeitpunkt, 24), badge: '24h' },
          { law: 'FINMA — Vollstaendige Meldung', deadline: computeDeadline(erkennungszeitpunkt, 72), badge: '72h' },
        )
      }
    }
    ```
