    **B. Create `src/components/wizard/WizardShell.tsx`** with `'use client'`:

    This is the top-level wizard container that:
    1. Wraps everything in `<WizardProvider>`
    2. Renders `<WizardProgress>` at the top
    3. Renders the current step component based on `state.currentStep`
    4. Renders `<StepNavigator>` at the bottom

    ```ts
    interface WizardShellProps {
      onComplete?: (state: WizardState) => void
    }
    ```

    **Inner component pattern** (because useWizard must be inside WizardProvider):

    ```tsx
    export function WizardShell({ onComplete }: WizardShellProps) {
      return (
        <WizardProvider>
          <WizardShellInner onComplete={onComplete} />
        </WizardProvider>
      )
    }

    function WizardShellInner({ onComplete }: WizardShellProps) {
      const { state, dispatch } = useWizard()

      const handleNext = () => {
        if (state.currentStep === MAX_STEP && onComplete) {
          onComplete(state)
        } else {
          dispatch({ type: 'NEXT_STEP' })
        }
      }

      const handlePrev = () => dispatch({ type: 'PREV_STEP' })

      // Step routing — placeholder components will be created in Plan 05
      const stepComponents: Record<number, React.ReactNode> = {
        0: <div>Placeholder: No-Go Interstitial</div>,
        1: <div>Placeholder: Einstieg</div>,
        2: <div>Placeholder: Erfassen</div>,
        3: <div>Placeholder: Klassifikation</div>,
        4: <div>Placeholder: Reaktion</div>,
        5: <div>Placeholder: Kommunikation</div>,
        6: <div>Placeholder: Dokumentation</div>,
      }

      return (
        <div className="max-w-2xl mx-auto px-4 py-8">
          <WizardProgress currentStep={state.currentStep} totalSteps={MAX_STEP} />
          <div className="min-h-[300px] py-6">
            {stepComponents[state.currentStep]}
          </div>
          <StepNavigator
            currentStep={state.currentStep}
            onNext={handleNext}
            onPrev={handlePrev}
            showPrev={state.currentStep > MIN_STEP}
            nextLabel={state.currentStep === MAX_STEP ? 'Abschliessen' : 'Weiter'}
          />
        </div>
      )
    }
    ```
